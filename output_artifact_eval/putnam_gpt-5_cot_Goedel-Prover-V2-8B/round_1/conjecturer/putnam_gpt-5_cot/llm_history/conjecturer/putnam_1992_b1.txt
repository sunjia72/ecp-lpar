\[
\frac{x_i+x_j}{2}\qquad (i\ne j).
\]

\[
x_1+x_2 < x_1+x_3 < \cdots < x_1+x_n < x_2+x_n < x_3+x_n < \cdots < x_{n-1}+x_n.
