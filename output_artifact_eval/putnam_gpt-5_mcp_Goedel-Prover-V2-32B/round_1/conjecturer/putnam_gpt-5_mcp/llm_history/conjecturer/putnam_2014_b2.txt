import math

def approx_max(n=400):
    # discretize [1,3] into n equal intervals, choose f_i in {-1,1} with zero sum => half +1 half -1
    xs=[1+(i+0.5)*2/n for i in range(n)]
    ws=[1/x for x in xs]
    # maximize sum f_i w_i with sum f_i = 0, |f_i|<=1 => choose +1 on largest weights
    idx=sorted(range(n), key=lambda i: ws[i], reverse=True)
    f=[-1]*n
    for i in idx[:n//2]:
        f[i]=1
    val=sum(f[i]*ws[i] for i in range(n))*(2/n)
