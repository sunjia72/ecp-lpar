from functools import lru_cache

def moves(mask, N):
    # return list of next masks (not including pass unless terminal)
    res=[]
    for i in range(N-1):
        if not (mask>>i)&1 and not (mask>>(i+1))&1:
            res.append(mask | (1<<i) | (1<<(i+1)))
    if not res:
        res=[mask]  # pass/self-loop terminal
    return res

def comp(mask,N):
    return N - mask.bit_count()

@lru_cache(None)
def val(mask,N,turn):
    # turn 0=maximizer (strategy), 1=minimizer
    ms=moves(mask,N)
    if ms==[mask]:
        return comp(mask,N)
    vals=[val(m,N,1-turn) for m in ms]
    return max(vals) if turn==0 else min(vals)
