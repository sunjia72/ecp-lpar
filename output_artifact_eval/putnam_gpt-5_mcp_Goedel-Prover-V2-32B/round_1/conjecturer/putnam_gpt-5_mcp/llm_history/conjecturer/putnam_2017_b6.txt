from itertools import permutations

def count(p,m):
    coeff=[1,1]+list(range(2,m+1))
    n=len(coeff)
    vals=list(range(p))
    cnt=[0]*p
    total=0
    for a in permutations(vals,n):
        s=sum(c*x for c,x in zip(coeff,a))%p
        cnt[s]+=1
        total+=1
