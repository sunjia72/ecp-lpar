def params(m):
    x=m
    u=0
    while x%2==0:
        u+=1; x//=2
    a=0
    while x%3==0:
        a+=1; x//=3
    b=0
    while x%5==0:
        b+=1; x//=5
    r=x
    return u,a,b,r

def phi(n):
    res=n
    x=n
    p=2
    while p*p<=x:
        if x%p==0:
            while x%p==0: x//=p
            res-=res//p
        p+=1
    if x>1: res-=res//x
    return res

def choose_n(m):
    u,a,b,r=params(m)
    if r==1 and a==0 and b==0:
