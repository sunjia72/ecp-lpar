\[
\sum_{k=1}^{p-1} k^d=0 \quad (\bmod p),
\]
so \(c_j=0\).

\[
\frac{1}{m!}k^{2m}=\frac{1}{m!}k^{p-1}.

\[
\sum_{k=1}^{p-1}k^{p-1}=p-1\equiv -1 \not\equiv 0 \pmod p.
\]
Thus
\[
c_m = -\frac1{m!}\ne 0.
\]
