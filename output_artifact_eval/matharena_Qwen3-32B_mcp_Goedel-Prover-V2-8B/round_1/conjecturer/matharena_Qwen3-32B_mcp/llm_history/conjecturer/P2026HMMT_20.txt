def is_valid_step(current, next_node):
    row1, col1 = current
    row2, col2 = next_node
    # Vertical step: same row, adjacent column
    vertical = (row1 == row2) and (abs(col1 - col2) == 1)
    # Horizontal step: same column, adjacent row
    horizontal = (col1 == col2) and (abs(row1 - row2) == 1)

def count_hamiltonian_paths(n_rows, n_cols, start_col, end_col):
    # Create the grid nodes
    nodes = [(row, col) for row in range(n_rows) for col in range(n_cols)]
    total = 0

    def dfs(current_path, visited):
        nonlocal total
        if len(current_path) == len(nodes):
            # Check if the last node is in the end_col
            if current_path[-1][1] == end_col:
                total += 1
            return
        # Get the current node
        current = current_path[-1]
        # Try all neighbors
        for neighbor in nodes:
            if neighbor not in visited and is_valid_step(current, neighbor):
                visited.add(neighbor)
                current_path.append(neighbor)
                dfs(current_path, visited)
                current_path.pop()
                visited.remove(neighbor)

    # Start from each node in start_col
    for start_row in range(n_rows):
        start = (start_row, start_col)
        dfs([start], {start})

    return total
