def count_by_segments(n=20):
    UNK = n  # pointer value n means undefined
    @lru_cache(None)
    def dp(cur, rem0, rem1, rem2, p0, p1, p2):
        rem = [rem0, rem1, rem2]
        p = [p0, p1, p2]
        if rem[cur] == 0:
            return 0
        total = 0
        start = p[cur]
        assert start != UNK
        # choose length of current segment
        for L in range(1, rem[cur] + 1):
            rem2a = rem.copy()
            pa = p.copy()
            endcol = (start + L - 1) % n
            pa[cur] = (start + L) % n
            rem2a[cur] -= L
            if rem2a == [0,0,0]:
                # can finish only if current row is 0
                if cur == 0:
                    total += 1
                continue
            # otherwise must vertical to adjacent row
            for nxt in ([cur-1] if cur==2 else ([cur+1] if cur==0 else [0,2])):
                # need nxt to have remaining vertices
                if rem2a[nxt] == 0:
                    continue
                if pa[nxt] == UNK:
                    pa2 = pa.copy(); pa2[nxt] = endcol
                    total += dp(nxt, rem2a[0], rem2a[1], rem2a[2], pa2[0], pa2[1], pa2[2])
                elif pa[nxt] == endcol:
                    total += dp(nxt, rem2a[0], rem2a[1], rem2a[2], pa[0], pa[1], pa[2])
        return total
    fixed = dp(2,20,20,20,UNK,UNK,0)
    return fixed
