from fractions import Fraction

def d(n):
    return Fraction(1, 2 ** ((n + 1)//2))

def a(n):
    if n <= 0:
        return Fraction(0,1)
    s = Fraction(0,1)
    for k in range(1, n+1):
        s += d(k)
    return s

def good(n, N=100):
    if n <= 0: return False
    for i in range(1, min(n, N-n+1)):
        if not (a(n-i) + a(n+i) < 2*a(n)):
            return False
    return True

# For several slopes s, compute argmax of a_n - s n over n=1..80.
slopes = [Fraction(1,2), Fraction(1,4), Fraction(1,8), Fraction(1,16), Fraction(1,32), Fraction(1,64)]
for s in slopes:
    vals = [(a(n) - s*n, n) for n in range(1,81)]
    M = max(v for v,_ in vals)
    # largest maximizer
    nstar = max(n for v,n in vals if v == M)
